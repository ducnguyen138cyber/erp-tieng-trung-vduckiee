  function patchQuizHeading(sectionNumber) {

  function sectionTitle(section, fallback) {
    var heading = section.querySelector(".hsk-section-title h4") || section.querySelector(".hsk-quiz-head h4") || section.querySelector("strong");
    return heading ? heading.textContent.replace(/^\s*\d+\.\s*/, "").trim() : fallback;
  }

  function rebuildMobileStepper() {
    var lesson = byId("hskLesson");
    if (!lesson) return;
    var oldStepper = byId("hskMobileStepper");
    if (oldStepper && oldStepper.parentNode) oldStepper.parentNode.removeChild(oldStepper);
    var oldBottom = lesson.querySelectorAll(".hsk-mobile-step-bottom");
    for (var i = 0; i < oldBottom.length; i++) oldBottom[i].parentNode.removeChild(oldBottom[i]);

    var sections = [], activeIndex = 0;
    for (var child = lesson.firstElementChild; child; child = child.nextElementSibling) {
      if (child.classList && child.classList.contains("hsk-section")) {
        if (child.classList.contains("mobile-active")) activeIndex = sections.length;
        sections.push(child);
      }
    }
    if (!sections.length) return;
    activeIndex = Math.max(0, Math.min(activeIndex, sections.length - 1));
    var titles = [];
    for (var s = 0; s < sections.length; s++) {
      sections[s].classList.toggle("mobile-active", s === activeIndex);
      titles.push(sectionTitle(sections[s], "Nội dung " + (s + 1)));
    }
    var stepper = document.createElement("div");
    stepper.className = "hsk-mobile-stepper";
    stepper.id = "hskMobileStepper";
    stepper.innerHTML = '<div class="hsk-mobile-step-meta"><strong>Phần ' + (activeIndex + 1) + "/" + sections.length + '</strong><span>' + escapeHtml(titles[activeIndex]) + '</span></div><div class="hsk-mobile-step-track"><i style="width:' + Math.round((activeIndex + 1) * 100 / sections.length) + '%"></i></div>';
    lesson.insertBefore(stepper, sections[0]);

    var completed = readJson(PROGRESS_KEY, {});
    var isLast = activeIndex === sections.length - 1;
    var item = lessons[activeLessonIndex()];
    var nextAction = isLast ? (completed[item.id] ? "next-lesson" : "section-complete") : "section-next";
    var nextLabel = isLast ? (completed[item.id] ? "Bài kế tiếp →" : "✓ Đánh dấu hoàn thành") : "Tiếp: " + titles[activeIndex + 1] + " →";
    var bottom = document.createElement("div");
    bottom.className = "hsk-mobile-step-bottom";
    bottom.innerHTML = '<button data-hsk-action="section-prev"' + (activeIndex === 0 ? " disabled" : "") + '>← Phần trước</button><button data-hsk-action="' + nextAction + '">' + escapeHtml(nextLabel) + "</button>";
    sections[activeIndex].appendChild(bottom);
  }

  function createFilterButton(value, label, active, attribute) {
    return '<button type="button" ' + attribute + '="' + escapeHtml(value) + '" class="' + (active ? "active" : "") + '">' + escapeHtml(label) + '</button>';
  }

  function renderSecondaryFilters() {
    var container = byId("dictionarySecondaryFilter");
    if (!container) return;
    var html = "";
    if (state.source === "hsk") {
      html += '<span class="dictionary-filter-label">Cấp độ</span>';
      html += createFilterButton("all", "Tất cả", state.hskLevel === "all", "data-dictionary-hsk-level");
      for (var level = 1; level <= 4; level++) html += createFilterButton(String(level), "HSK " + level, String(state.hskLevel) === String(level), "data-dictionary-hsk-level");
    } else if (state.source === "erp") {
      var erpFilters = [
        ["all", "Tất cả"], ["production", "Sản xuất"], ["warehouse", "Kho"],
        ["purchasing", "Mua hàng"], ["accounting", "Kế toán"],
        ["quality", "Chất lượng"], ["other", "Khác"]
      ];
      html += '<span class="dictionary-filter-label">Nghiệp vụ</span>';
      for (var i = 0; i < erpFilters.length; i++) html += createFilterButton(erpFilters[i][0], erpFilters[i][1], state.erpGroup === erpFilters[i][0], "data-dictionary-erp-group");
    }
    container.innerHTML = html;
    container.classList.toggle("hidden", state.source === "all");
  }

  function meaningHtml(record) {
    var hskMeaning = record.meanings.hsk.join("; ");
    var erpMeaning = record.meanings.erp.join("; ");
    if (state.source === "hsk") return escapeHtml(hskMeaning || erpMeaning || "Chưa có nghĩa tiếng Việt");
    if (state.source === "erp") return escapeHtml(erpMeaning || hskMeaning || "Chưa có nghĩa tiếng Việt");
    if (hskMeaning && erpMeaning && normalizeSearch(hskMeaning) !== normalizeSearch(erpMeaning)) {
      return '<span><b>HSK:</b> ' + escapeHtml(hskMeaning) + '</span><span><b>ERP:</b> ' + escapeHtml(erpMeaning) + '</span>';
    }
    return escapeHtml(hskMeaning || erpMeaning || "Chưa có nghĩa tiếng Việt");
  }

  function badgeHtml(record) {
    var html = "";
    if (record.sources.indexOf("hsk") !== -1) html += '<span class="dictionary-badge badge-hsk">HSK' + (record.hskLevels.length ? " " + record.hskLevels.join("/") : "") + '</span>';
    if (record.sources.indexOf("erp") !== -1) html += '<span class="dictionary-badge badge-erp">ERP</span>';
    var categories = record.erpCategories.slice(0, 2);
    for (var i = 0; i < categories.length; i++) html += '<span class="dictionary-badge badge-category">' + escapeHtml(categories[i]) + '</span>';
    if (record.erpCategories.length > 2) html += '<span class="dictionary-badge badge-category">+' + (record.erpCategories.length - 2) + '</span>';
    return html;
  }

  function exampleHtml(record) {
    var list = state.source === "hsk" ? record.examples.hsk : state.source === "erp" ? record.examples.erp : record.examples.hsk.concat(record.examples.erp);
    var example = list[0];
    if (!example || (!example.zh && !example.vi)) return "";
    return '<div class="dictionary-example"><strong>' + escapeHtml(example.zh) + '</strong><span>' + escapeHtml(example.vi) + '</span></div>';
  }

  function renderCard(record) {
    return '<article class="dictionary-card" data-word="' + escapeHtml(record.hanzi) + '">' +
      '<div class="dictionary-card-top"><div class="dictionary-word"><strong>' + escapeHtml(record.hanzi) + '</strong><span>' + escapeHtml(record.pinyin || "—") + '</span></div><button class="dictionary-speak" type="button" data-dictionary-speak="' + escapeHtml(record.hanzi) + '" aria-label="Nghe phát âm ' + escapeHtml(record.hanzi) + '">♪ Nghe</button></div>' +
      '<div class="dictionary-badges">' + badgeHtml(record) + '</div>' +
      (record.nearVi ? '<p class="dictionary-near">Đọc gần: <strong>' + escapeHtml(record.nearVi) + '</strong></p>' : "") +
      '<div class="dictionary-meaning">' + meaningHtml(record) + '</div>' +
      exampleHtml(record) +
      '</article>';
  }

  function renderResults() {
    if (!dictionaryPage) return;
    var filtered = filterRecords(records, state);
    var shown = Math.min(filtered.length, state.limit);
    var html = "";
    for (var i = 0; i < shown; i++) html += renderCard(filtered[i]);
    if (!html) html = '<div class="dictionary-empty">Không tìm thấy từ phù hợp. Hãy thử bỏ bớt bộ lọc hoặc tìm bằng một phần của từ.</div>';
    byId("dictionaryResults").innerHTML = html;
    byId("dictionaryCount").textContent = filtered.length;
    byId("dictionaryAllCount").textContent = sourceCount("all");
    byId("dictionaryHskCount").textContent = sourceCount("hsk");
    byId("dictionaryErpCount").textContent = sourceCount("erp");
    byId("dictionarySummary").textContent = "Có " + records.length + " bản ghi duy nhất · " + sourceCount("hsk") + " từ HSK · " + sourceCount("erp") + " từ ERP.";
    var more = byId("dictionaryMore");

        .concat(record.meanings.erp, record.meanings.hsk, record.erpCategories, record.notes);
      record.searchText = normalizeSearch(searchParts.join(" "));
      record.searchCompact = record.searchText.replace(/\s+/g, "");
    }
    return records;
  }

  function filterRecords(records, state) {
    state = state || {};
    var source = state.source || "all";
    var query = normalizeSearch(state.query || "");
    var queryCompact = query.replace(/\s+/g, "");
    var hskLevel = state.hskLevel || "all";
    var erpGroup = state.erpGroup || "all";
    return (records || []).filter(function (record) {
      if (source !== "all" && record.sources.indexOf(source) === -1) return false;
      if (source === "hsk" && hskLevel !== "all" && record.hskLevels.indexOf(Number(hskLevel)) === -1) return false;
      if (source === "erp" && erpGroup !== "all" && record.erpGroups.indexOf(erpGroup) === -1) return false;
      return !query || record.searchText.indexOf(query) !== -1 || record.searchCompact.indexOf(queryCompact) !== -1;
    });
  }

  function currentHskWordCount() {
    var total = 0;
    var curriculum = root.HSKCurriculum;
    if (!curriculum || !curriculum.levels) return total;
    for (var level = 1; level <= 4; level++) {
      var lessons = curriculum.levels[level] || [];
      for (var i = 0; i < lessons.length; i++) total += (lessons[i].words || []).length;
    }
    return total;
  }

  function dataSignature() {
    return [
      (root.ERP_TERMS || []).length,
      currentHskWordCount(),
      root.HSKCurriculum && root.HSKCurriculum.standard || "",
      root.ERP_TERMS_V77 && root.ERP_TERMS_V77.totalCount || ""
    ].join("|");
  }

  function refreshData(force) {
    var signature = dataSignature();
    if (!force && signature === currentSignature && records.length) return false;
    records = buildRecords(root.ERP_TERMS || [], root.HSKCurriculum || {}, root.ERPPronunciation);
    currentSignature = signature;
    if (root.VDuckieDictionary) root.VDuckieDictionary.records = records;
    return true;
  }

  function sourceCount(source) {
    if (source === "all") return records.length;
    var count = 0;
    for (var i = 0; i < records.length; i++) if (records[i].sources.indexOf(source) !== -1) count++;
    return count;
  }
